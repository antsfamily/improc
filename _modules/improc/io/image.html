[docs]def imreadadv(filepath, verbose=False):
    """read image data from a file

    read image data from a file

    Parameters
    ----------
    filepath : {string}
        image file path
    verbose : {bool}, optional
        show more information (the default is False.)

    Returns
    -------
    numpy array
        image data array.
    """

    if verbose:
        logging.info("---In imreadadv...")

    ext = os.path.splitext(filepath)[-1]

    if ext in ['.tif', '.tiff', '.TIF', '.TIFF']:
        A = tifread(filepath)
    else:
        A = scipyimread(filepath)

    if verbose:
        print(A.shape, filepath)
        logging.info("---Out imreadadv.")

    return A

        tifsave(filepath, A)
    else:
        A = scipyimsave(filepath, A)
    # logging.info("---Out imwriteadv.")

    return 0
